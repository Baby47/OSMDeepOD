\subsection{RQ}
RQ\footnote{\url{http://python-rq.org/}} (Redis Queue) ist eine einfache Library f√ºr Python um Jobs in Redis einzureihen.

\subsubsection{Beispiel Queue}
\begin{python}
from rq import Queue, use_connection
from redis import Redis

class RQueue:
   def __init__(self):
       redis = Redis(ip, port, password)
       q = Queue(connection=redis)
       job = q.enqueue(self.add, 2, 3)
       print job.result

   def add(self, x, y):
       return x + y
\end{python}

from redis import Redis
